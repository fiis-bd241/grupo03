-- vista de Rendimiento de Usuarios
CREATE OR REPLACE VIEW public."RendimientoUsuarios" AS
SELECT 
    e.id_empleado,
    e.nombre AS empleado,
    r.nombre_rol AS rol,
    t.fecha_fin AS fecha,
    COUNT(t.id_tarea) AS tareas_completadas,
    AVG(COALESCE(EXTRACT(EPOCH FROM (t.fecha_fin_real::timestamp - t.fecha_inicio::timestamp)), EXTRACT(EPOCH FROM (CURRENT_DATE::timestamp - t.fecha_inicio::timestamp))) / 3600) AS tiempo_promedio,
    AVG(t.calidad) AS calidad_trabajo,
    COUNT(DISTINCT pr."Reunion_Id") AS participacion_reuniones
FROM 
    public."Empleado" e
JOIN 
    public."Tarea" t ON e.id_empleado = t.id_empleado
LEFT JOIN 
    public."Roles" r ON e.rol_id = r.id_rol
LEFT JOIN 
    public."Participa_en" pr ON e.id_empleado = pr."Participante_Id"
GROUP BY 
    e.id_empleado, e.nombre, r.nombre_rol, t.fecha_fin;
